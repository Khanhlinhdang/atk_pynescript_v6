import pandas as pd

from source import indicator, input, plot, ta


# indicator(...) declares that this module is an analysis script, not an execution script.
# overlay=True means the output belongs on the price chart instead of a separate panel.
# max_bars_back should cover the longest lookback the script needs.
indicator("Annotated EMA Overlay", overlay=True, precision=4, max_bars_back=200)

# input.int is used for discrete lookback counts.
# key is the stable runtime identity for the setting.
length = input.int(34, title="EMA Length", key="length", minval=1)

# input.source is better than a plain string when the user is selecting a market source.
source_type = input.source("close", title="Source", key="source_type")

# plot declares how to draw a column that will exist later in build_indicator_frame.
# The first argument is the output column name, not the formula.
plot("ema_value", key="ema_value", title="EMA", color="#00c853", width=2)


def build_indicator_frame(df: pd.DataFrame, params: dict | None = None) -> pd.DataFrame:
    # Always copy and reset_index so the builder works with a clean, row-aligned frame.
    frame = df.copy().reset_index(drop=True)

    # Merge runtime params over import-time defaults.
    merged = {
        "length": int(length),
        "source_type": str(source_type),
    } | dict(params or {})

    # Resolve the selected source safely. If the requested source is missing, fall back to close.
    source_name = str(merged.get("source_type", "close") or "close")
    source_series = frame[source_name] if source_name in frame.columns else frame["close"]

    # All TA work belongs here, never in plot(...) and never in build_visuals(...).
    frame["ema_value"] = ta.ema(source_series, int(merged.get("length", 34) or 34))
    return frame
import numpy as np
import pandas as pd

from source import strategy, input, ta, build_mapped_trade_frame


# strategy(...) is chosen because this script will emit canonical trade-frame execution fields.
# process_orders_on_close is the most practical timing flag in the current runtime.
strategy("Annotated EMA Strategy", overlay=True, process_orders_on_close=True, max_bars_back=240)
fast_period = input.int(10, title="Fast EMA", key="fast_period", minval=1)
slow_period = input.int(24, title="Slow EMA", key="slow_period", minval=2)
trade_qty = input.float(1.0, title="Trade Qty", key="trade_qty", minval=0.0, step=0.1)


def build_signal_frame(df: pd.DataFrame, params: dict | None = None) -> pd.DataFrame:
    frame = df.copy().reset_index(drop=True)
    merged = {
        "fast_period": int(fast_period),
        "slow_period": int(slow_period),
        "trade_qty": float(trade_qty),
    } | dict(params or {})

    # Signal logic lives here.
    ema_fast = ta.ema(frame["close"], int(merged.get("fast_period", 10) or 10))
    ema_slow = ta.ema(frame["close"], int(merged.get("slow_period", 24) or 24))
    frame["buy_signal"] = ta.crossover(ema_fast, ema_slow).fillna(False)
    frame["sell_signal"] = ta.crossunder(ema_fast, ema_slow).fillna(False)

    # entry_side is the execution direction.
    frame["entry_side"] = np.where(frame["buy_signal"], "BUY", np.where(frame["sell_signal"], "SELL", ""))

    # entry_price should reflect how the strategy intends to enter.
    frame["entry_price"] = frame["open"]

    # quantity is used because this strategy sizes in absolute units.
    frame["quantity"] = float(merged.get("trade_qty", 1.0) or 0.0)
    frame["size_pct"] = 0.0
    return frame


def build_trade_frame(signal_df: pd.DataFrame, params: dict | None = None, styles: dict | None = None) -> pd.DataFrame:
    # Keep trade mapping thin. Do not compute fresh TA or signals here.
    return build_mapped_trade_frame(signal_df)

Static Visual Recipes with Inline Explanations

Why start here

If a visual can be expressed as one value or marker per row, static declarations are still the cleanest authoring path.

What to watch for

The declaration names the output column and style. The frame builder must still create the actual data columns.

import pandas as pd

from source import indicator, input, linefill, plot, ta


indicator("Annotated BBands Fill", overlay=True, max_bars_back=240)
length = input.int(20, title="Length", key="length", minval=2)
multiplier = input.float(2.0, title="StdDev Mult", key="multiplier", minval=0.1, step=0.1)
mamode = input.string("sma", title="MA Mode", key="mamode")
source_type = input.source("close", title="Source", key="source_type")

# These three plot declarations do not calculate bands.
# They only say: if the frame later contains columns lb, cb, and ub, draw them like this.
plot("lb", key="bb_lower", title="Lower Band", color="#ff0000", width=1)
plot("cb", key="bb_basis", title="Basis", color="#ffa500", width=1)
plot("ub", key="bb_upper", title="Upper Band", color="#00aa00", width=1)

# linefill.new links two existing line identities by key.
# That is why the first two arguments are plot keys, not dataframe column names.
linefill.new("bb_lower", "bb_upper", key="bb_band_fill", color="rgba(63,57,100,0.44)")


def build_indicator_frame(df: pd.DataFrame, params: dict | None = None) -> pd.DataFrame:
    frame = df.copy().reset_index(drop=True)
    merged = {
        "length": int(length),
        "multiplier": float(multiplier),
        "mamode": str(mamode),
        "source_type": str(source_type),
    } | dict(params or {})

    source_name = str(merged.get("source_type", "close") or "close")
    source_series = frame[source_name] if source_name in frame.columns else frame["close"]

    # ta.bbands returns upper, basis, lower in one call.
    upper, basis, lower = ta.bbands(
        source_series,
        length=max(int(merged.get("length", 20) or 20), 2),
        std=float(merged.get("multiplier", 2.0) or 2.0),
        mamode=str(merged.get("mamode", "sma") or "sma").lower(),
    )

    # These column names must match the plot source names declared above.
    frame["lb"] = lower.reset_index(drop=True)
    frame["cb"] = basis.reset_index(drop=True)
    frame["ub"] = upper.reset_index(drop=True)
    return frame

import pandas as pd

from source import indicator, ta


indicator("Annotated Object Visual", overlay=True, max_bars_back=200)


def build_indicator_frame(df: pd.DataFrame, params: dict | None = None) -> pd.DataFrame:
    frame = df.copy().reset_index(drop=True)
    frame["basis"] = ta.ema(frame["close"], 20)

    if not frame.empty:
        last = frame.iloc[-1]
        anchor = frame.iloc[max(len(frame) - 10, 0)]

        # frame.attrs stores static object config.
        # This is safe because the payload is describing the object, not caching the full frame.
        frame.attrs["visual_payload"] = {
            "line": {
                "x1": int(anchor["index"]),
                "y1": float(anchor["close"]),
                "x2": int(last["index"]),
                "y2": float(last["close"]),
                "color": "#ffd600",
                "text": "TREND",
            },
            "label": {
                "x": int(last["index"]),
                "y": float(last["close"]),
                "text": "LAST",
                "style": "label_down",
                "color": "#2962ff",
            },
        }
    else:
        frame.attrs["visual_payload"] = {}
    return frame


def build_visuals(frame: pd.DataFrame, params: dict | None = None, ctx=None):
    # build_visuals is render-only. No TA, no rolling windows, no signal logic here.
    if frame is None or frame.empty or ctx is None:
        return None

    payload = dict(frame.attrs.get("visual_payload") or {})
    if not payload:
        return None

    ctx.label.new(key="last_label", **dict(payload.get("label") or {}))
    return ctx.line.new(key="trend_line", **dict(payload.get("line") or {}))

def build_indicator_frame(df: pd.DataFrame, params: dict | None = None) -> pd.DataFrame:
    frame = df.copy().reset_index(drop=True)
    frame["ema_fast"] = ta.ema(frame["close"], 10)
    frame["ema_slow"] = ta.ema(frame["close"], 21)

    if not frame.empty:
        last = frame.iloc[-1]
        trend_state = "UP" if float(last["ema_fast"]) >= float(last["ema_slow"]) else "DOWN"
        frame.attrs["visual_table_payload"] = {
            "position": "top-left",
            "title": "Signal Summary",
            "columns": 2,
            "rows": 3,
            "cells": [
                {"column": 0, "row": 0, "text": "Metric", "text_color": "#ffffff", "bgcolor": "#0f172a"},
                {"column": 1, "row": 0, "text": "Value", "text_color": "#ffffff", "bgcolor": "#0f172a"},
                {"column": 0, "row": 1, "text": "Close"},
                {"column": 1, "row": 1, "text": f"{float(last['close']):.2f}", "text_color": "#00c853"},
                {"column": 0, "row": 2, "text": "Trend"},
                {"column": 1, "row": 2, "text": trend_state, "text_color": "#ffd600"},
            ],
        }
    else:
        frame.attrs["visual_table_payload"] = {}
    return frame


def build_visuals(frame: pd.DataFrame, params: dict | None = None, ctx=None):
    payload = dict(frame.attrs.get("visual_table_payload") or {})
    if not payload:
        return None

    # ctx.atk.table is chosen because a dashboard is a composite ATK-native renderer,
    # not a simple row-aligned line or marker.
    return ctx.atk.table(key="annotated_dashboard", **payload)

# @name: pynescript_strategy_starter
import numpy as np
import pandas as pd

from source import strategy, input, ta, build_mapped_trade_frame

strategy("Pyne Strategy Starter", overlay=True, process_orders_on_close=True)
fast_period = input.int(10, title="Fast", key="fast_period")
slow_period = input.int(21, title="Slow", key="slow_period")
trade_qty = input.float(1.0, title="Trade Qty", key="trade_qty")
size_pct = input.float(0.0, title="Size %", key="size_pct")


def build_signal_frame(df: pd.DataFrame, params: dict | None = None) -> pd.DataFrame:
    frame = df.copy().reset_index(drop=True)
    p = {
        "fast_period": int(fast_period),
        "slow_period": int(slow_period),
        "trade_qty": float(trade_qty),
        "size_pct": float(size_pct),
    } | dict(params or {})
    ema_fast = ta.ema(frame["close"], int(p["fast_period"]))
    ema_slow = ta.ema(frame["close"], int(p["slow_period"]))
    frame["ema_fast"] = ema_fast
    frame["ema_slow"] = ema_slow
    frame["buy_signal"] = ta.crossover(ema_fast, ema_slow).fillna(False)
    frame["sell_signal"] = ta.crossunder(ema_fast, ema_slow).fillna(False)
    frame["entry_side"] = np.where(frame["buy_signal"], "BUY", np.where(frame["sell_signal"], "SELL", ""))
    frame["entry_price"] = frame["open"]
    frame["quantity"] = float(p.get("trade_qty", 0.0) or 0.0)
    frame["size_pct"] = float(p.get("size_pct", 0.0) or 0.0)
    return frame


def build_trade_frame(signal_df: pd.DataFrame, params: dict | None = None, styles: dict | None = None) -> pd.DataFrame:
    return build_mapped_trade_frame(signal_df)

def build_visuals(frame: pd.DataFrame, params: dict | None = None, ctx=None):
    if frame is None or frame.empty or ctx is None:
        return None
    payload = dict(frame.attrs.get("visual_indicator_payload") or {})

Multi-timeframe

MTF Confirmation Strategy

Download source Open MTF handbook Open examples hub

When to use it

Use this as the baseline example for higher-timeframe confirmation, especially when a local entry signal must be gated by a broader trend filter.

What it teaches

request.security belongs in build_signal_frame with the rest of the strategy calculation. The mapped trade frame remains a thin wrapper.

# @name: pynescript_mtf_confirmation_strategy
import numpy as np
import pandas as pd

from source import strategy, input, ta, request, build_mapped_trade_frame

strategy("Pyne MTF Confirmation Strategy", overlay=True, process_orders_on_close=True, max_bars_back=180)
fast_period = input.int(10, title="Fast EMA", key="fast_period")
slow_period = input.int(24, title="Slow EMA", key="slow_period")
trend_length = input.int(34, title="HTF Trend EMA", key="trend_length")
confirm_tf = input.timeframe("15m", title="Confirm TF", key="confirm_tf")
trade_qty = input.float(1.0, title="Trade Qty", key="trade_qty")


def build_signal_frame(df: pd.DataFrame, params: dict | None = None) -> pd.DataFrame:
    frame = df.copy().reset_index(drop=True)
    p = {
        "fast_period": int(fast_period),
        "slow_period": int(slow_period),
        "trend_length": int(trend_length),
        "confirm_tf": str(confirm_tf),
    } | dict(params or {})
    ema_fast = ta.ema(frame["close"], int(p["fast_period"]))
    ema_slow = ta.ema(frame["close"], int(p["slow_period"]))
    htf_trend = request.security(
        frame,
        str(p["confirm_tf"]),
        lambda x: ta.ema(x["close"], int(p["trend_length"])),
    )
    frame["ema_fast"] = ema_fast
    frame["ema_slow"] = ema_slow
    frame["htf_trend"] = htf_trend
    frame["buy_signal"] = ta.crossover(ema_fast, ema_slow).fillna(False) & (frame["close"] >= htf_trend.fillna(frame["close"]))
    frame["sell_signal"] = ta.crossunder(ema_fast, ema_slow).fillna(False) & (frame["close"] <= htf_trend.fillna(frame["close"]))
    frame["entry_side"] = np.where(frame["buy_signal"], "BUY", np.where(frame["sell_signal"], "SELL", ""))
    frame["entry_price"] = frame["open"]
    frame["quantity"] = float(trade_qty)
    frame["size_pct"] = 0.0
    return frame


def build_trade_frame(signal_df: pd.DataFrame, params: dict | None = None, styles: dict | None = None) -> pd.DataFrame:
    return build_mapped_trade_frame(signal_df)

Volume Profile Fixed Range

# @name: pynescript_volume_profile_fixed_range
import pandas as pd

from source.controls.tradingview.volume_profile_fixed_range import calculate_volume_profile
from source import indicator, input


indicator("Pyne Volume Profile Fixed Range", overlay=True, max_bars_back=900)
look_back_bar = input.int(500, title="Look Back Bars", key="look_back_bar")
row_size = input.int(24, title="Row Size", key="row_size")
value_area_volume_percent = input.int(70, title="Value Area %", key="value_area_volume_percent")


def build_indicator_frame(df: pd.DataFrame, params: dict | None = None) -> pd.DataFrame:
    frame = df.copy().reset_index(drop=True)
    merged = {
        "look_back_bar": int(look_back_bar),
        "row_size": int(row_size),
        "value_area_volume_percent": int(value_area_volume_percent),
    } | dict(params or {})
    try:
        payload = calculate_volume_profile(
            frame.copy().reset_index(drop=True),
            max(int(merged.get("look_back_bar", 500) or 500), 20),
            max(int(merged.get("row_size", 24) or 24), 2),
            max(int(merged.get("value_area_volume_percent", 70) or 70), 1),
        )
    except Exception:
        payload = {}
    frame.attrs["volume_profile_payload"] = dict(payload or {})
    return frame


def build_visuals(frame: pd.DataFrame, params: dict | None = None, ctx=None):
    payload = dict(frame.attrs.get("volume_profile_payload") or {})
    return ctx.atk.volume_profile(
        payload,
        key="dual_volume_profile",
        orient="right",
        up_color="#0126A0",
        down_color="#ffa32b",
        value_area_title="VA",
        poc_title="POC",
    )
